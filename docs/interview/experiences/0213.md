```js
async function asyncPool(limit, arr, fn) {
  var exe = [];
  var res = [];
  for (let i of arr) {
    var p1 = Promise.resolve().then(() => fn(i));
    res.push(p1);
```

```js
// 节流函数 稀释 跟防抖的区别是
// 防抖是 多次触发只执行一次
// 节流是 N时间段内只执行一次
var throttle = (fn, waiter) => {
  let timer = null;
  return (...arg) => {
    if (timer) return;
    timer = setTimeout(() => {
      fn.apply(this, arg);
      timer = null;
    }, waiter);
  };
};
var throttle1 = (fn, delay) => {
  var t1;
  return (...args) => {
    var t2 = new Date();
    if (t2 - t1 > delay) {
      fn.apply(this, args);
      t1 = t2;
    }
  };
};
// 三秒之后才能执行 如果又动了一次 必须在等三秒
var debounce = (fn, waiter) => {
  var timer = null;
  return (...args) => {
    if (timer) {
      clearTimeout(timer);
    }
    timer = setTimeout(() => {
      fn.apply(this, args);
    }, waiter);
  };
};
```

```js
function remove(str) {
  var m = new Map()
  for (let i of str) {
    if (m.has(i)) {
      const len = m.get(i)
      m.set(i, len+1)
    } else {
      m.set(i, 1)
    }
  }
  // 3 2 1
  let min = Infinity
  for (let [i, item] of m.entries()) {
    if (Math.min(min, item)) {
      min = item
    }
  }
  let select = str[0]
  for (let [i, item] of m.entries()) {
    if (item===min) {
      select = i
    }
  }
  let res = ''

    if (i !== select) {

    }
  }
```
