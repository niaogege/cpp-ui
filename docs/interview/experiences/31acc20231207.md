- 输入 aaabbbbccbbcccccc，返回 { string: 'c', count: 6} \*/
function maxContinuousString(str) {
  if (!str) return
  let arr = [str[0]]
  let res = {
    string: '',
    count: 0
  }
  for (let i=1;i<str.length;i++) {
    let cur = str[i]
    if (!arr.includes(cur)) {
      arr = []
    }
    arr.push(cur)
    res.count = Math.max(res.count, arr.length)
    if (res.count === arr.length) {
      res.string = arr[0]
    }
  }
  return res
}
maxContinuousString('aaabbbbccbbcccccc')
