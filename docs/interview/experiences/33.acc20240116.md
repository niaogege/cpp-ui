```js
// dfs
// 前序 根左右
// 中序 左根右
// 后序 左右根
function preOrder(arr) {
  if (!arr) return [];
  let res = [];
  let dfs = (node) => {
    if (node) {
      res.push(node.val);
      dfs(node.left);
      dfs(node.right);
    }
  };
}
// BFS 前序
function bfspreOrder(root) {
  let stack = [root];
  let res = [];
  while (stack.length) {
    const last = stack.pop();
    res.push(last.val);
    // 右入栈
    if (last.right) {
      stack.push(last.right);
    }
    // 左入栈
    if (last.left) {
      stack.push(last.left);
    }
  }
  return res;
}
// postOrder 后序遍历
function postOrder(root) {
  let stack = [root];
  let res = [];
  while (stack.length) {
    const last = stack.pop();
    res.push(last.val);
    // 左入栈
    if (last.left) {
      stack.push(last.left);
    }
    // 右入栈
    if (last.right) {
      stack.push(last.right);
    }
  }
  return res.reverse();
}

// inorder 中序遍历 左根右
function inOrder(root) {
  let stack = [];
  let res = [];
  let cur = root;
  while (stack.length || cur) {
    if (cur) {
      stack.push(cur);
      cur = cur.left; // 左
    } else {
      cur = stack.pop();
      res.push(cur.val); // 根
      cur = cur.right; // 右
    }
  }
  return res;
}
```

- 写一个 Promise.all

```js
Promise.prototype.all = function (arr) {
  return new Promise((resolve, reject) => {
    let ans;
    for (let [k, val] of Object.entries(arr)) {
      Promise.resolve(val).then(
        (res) => {
          if (k + 1 == arr.length) {
            resolve(ans);
          }
          ans[k] = res;
        },
        (err) => {
          reject(err);
        },
      );
    }
  });
};
```
