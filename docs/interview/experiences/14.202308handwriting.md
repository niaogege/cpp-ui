### 9.once 函数 跟 memo 函数差不多

```js
function memo(fn) {
  var cache = {}
  return (...rest) => {
    var args=JSON.stringify(rest)
    return cache[args] || cache[args] = fn.apply(this, rest)
  }
}
const once = (fn) => {
  let res, isFirst = true
  return function (...args) {
    if (!isFirst) return res
    res = fn.call(this, ...args)
    isFirst = false
    return res
  }
}
const f = (x) => x;
const onceF = once(f);
//=> 3
onceF(3);
//=> 3
onceF(4);
```
